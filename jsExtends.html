js对象的几种创建方式

工厂模式

function createperson(name,age,job){
var o = new Object();
o.name = name;
o.age = age;
o.job = job;
return o;
}
var nicholas = createperson('nicholas',25,'software engineer');

构造函数模式

function Person(name,age,job){
this.name = name;
this.age = age;
this.job = job;
}
var nicholas = new Person('nicholas',25,'software engineer');
